# These are the SQL codes we used

```
CREATE TABLE  "CUSTOMERS" 
   (	"CUSTID" CHAR(3) NOT NULL ENABLE, 
	"CUSTNAME" VARCHAR2(50) NOT NULL ENABLE, 
	"CUSTADDRESS" VARCHAR2(99) NOT NULL ENABLE, 
	"CUSTCONTACT" VARCHAR2(25) NOT NULL ENABLE, 
	 CONSTRAINT "CUSTOMERS_PK" PRIMARY KEY ("CUSTID") ENABLE
   );

CREATE TABLE  "EMPLOYEE" 
   (	"EMPID" CHAR(1), 
	"EMPNAME" VARCHAR2(50), 
	"EMPCONTACT" VARCHAR2(25), 
	"EMPBACKUPCONTACT" VARCHAR2(25), 
	"EMPADDRESS" VARCHAR2(99), 
	"EMPGENDER" CHAR(1), 
	"EMPUSERNAME" VARCHAR2(15) NOT NULL ENABLE, 
	"EMPPASSWORD" VARCHAR2(8) NOT NULL ENABLE, 
	 CONSTRAINT "EMPLOYEE_PK" PRIMARY KEY ("EMPID") ENABLE
   );
CREATE TABLE  "PRODUCT" 
   (	"PRODID" CHAR(3), 
	"PRODNAME" VARCHAR2(50), 
	"INSTOCK" NUMBER, 
	"ARVDATE" DATE, 
	"EXPDATE" DATE, 
	"PRICE" NUMBER, 
	"SUPID" CHAR(3), 
	 CONSTRAINT "PRODUCT_PK" PRIMARY KEY ("PRODID") ENABLE, 
	 CONSTRAINT "PRODUCT_FK" FOREIGN KEY ("SUPID")
	  REFERENCES  "SUPPLIERS" ("SUPID") ENABLE
   );
CREATE TABLE  "SUPPLIERS" 
   (	"SUPID" CHAR(3), 
	"SUPNAME" VARCHAR2(50), 
	"SUPADDRESS" VARCHAR2(99), 
	"SUPCONTACT" VARCHAR2(25), 
	 CONSTRAINT "SUPPLIERS_PK" PRIMARY KEY ("SUPID") ENABLE
   );
```
